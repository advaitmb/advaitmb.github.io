\sectionTitle{Select Research Projects}{}
\begin{projects}

\project
	{Examining the Influence of Opinionated Language Models during Writing}{2022 -- 2023}
	{
	    \textit{sTech Lab, Cornell Tech | Advisors:~\href{https://tech.cornell.edu/people/mor-naaman/}{Prof. Mor Naaman}, \href{https://mauricejakesch.com/}{Prof. Maurice Jakesch}}
	}
	{\begin{itemize}
	\setlength\itemsep{0.3em}
     \item Studied how opinionated language models influence what writers express and believe through a large-scale co-writing experiment with 1,506 participants.
     \item Built a Reddit-like writing interface and assistant for controlled co-writing studies with opinionated language models.
     \item Ran a mixed-method follow-up study with interaction log analysis and retrospective interviews to examine mechanisms of AI influence during writing.
     \end{itemize}}

\project
	{Integrating Learning and Skilling with Data Annotation}{2023}
	{
	     \textit{Microsoft Research India | Advisor:~\href{https://www.microsoft.com/en-us/research/people/visesha/}{Dr. Vivek Seshadri}}
	}
	{\begin{itemize}
	\setlength\itemsep{0.3em}
     \item Developed a learn-and-earn framework that integrates financial literacy instruction with speech data generation on the Karya platform.
     \item Designed a story-based Marathi curriculum for a low-income stone quarry worker community in Wagholi, near Pune.
     \item Evaluated the two-week intervention with 55 participants, finding a 41.6\% improvement in financial literacy alongside high-quality labeled speech data generation.
     \end{itemize}}

\project
    {Studying Writer Interaction with Language Model-powered Writing Assistants}{2020 -- 2022}
		{ \textit{IIT Bombay | Advisor:~\href{http://www.idc.iitb.ac.in/~anirudha/}{Prof. Anirudha Joshi}}}
		{\begin{itemize}
		\setlength\itemsep{0.3em}
     \item Studied how writers interact with next-phrase suggestion systems and how those systems influence cognitive writing processes.
     \item Designed and built a GPT-2-powered predictive text editor for think-aloud and retrospective-protocol writing studies.
     \item Proposed a model of writer-suggestion interaction, including how suggestion systems reshape planning, attention, and writing flow.
     \end{itemize}}


\end{projects}
